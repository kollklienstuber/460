POV-RAY Python

A previously created POV-Ray object is centered and a "camera" pans throughout lanscrape to create a scence

import math #from labpdf fin = open('C:/base.pov') sdl = fin.read() # Read the entire file into a string fin.close() #finding camera locationVal = sdl[sdl.find('camera'):] #traverse through to find < and > of an object initial = locationVal.find('<') end = locationVal.find('>') #Location, X, Y, Z points assigned locationVal = locationVal[initial:end+1] xLoc = locationVal[1:locationVal.find(',')] newCoordVal = locationVal[locationVal.find(',') + 1:] yLoc = newCoordVal[0:newCoordVal.find(',')] newCoordVal = newCoordVal[newCoordVal.find(','):] zLoc = newCoordVal[1:newCoordVal.find('>')] count = 0 #count value effects length while count < 111: #formula for helix shape xLoc = float(xLoc) xLoc = math.cos(xLoc * (3.14/180)) yLoc = float(yLoc) yLoc = yLoc + 4 zLoc = float(zLoc) zLoc = math.sin(zLoc * (3.14/180)) MovedCamera = ('camera { location <&d,%d,%d> look_at <0,0,0> } \ light_source { <15,15,-20>, color rgb <0.90,0.90,0.85> }' % (yLoc,xLoc,zLoc)) newSDL = sdl + MovedCamera num = str(count) #to write a new file using the string sdl_new outfile = ( num + 'tmp.pov') fout = open( outfile, 'w') fout.write(newSDL) fout.close() import os pov_cmd = 'C:/bin/pvengine64.exe /EXIT +I%s +O%s -D -V +A +H600 +W800' cmd = pov_cmd % (outfile, num + 'tmp.png' ) os.system(cmd) count = count + 1 #Now make the movie print 'Encoding movie' os.system('mencoder.exe mf://*tmp.png -mf type=png:fps=20 -ovc lavc -lavcopts \ vcodec=msmpeg4:vbitrate=2160000:keyint=5:vhq -o movie.avi ' )
